// create table for store cartificate student list.
CREATE TABLE tb_cartificate_tmp (
id int(11) AUTO_INCREMENT,
reg_number varchar(150) UNIQUE,
name varchar(250),
father_name varchar(250),
address varchar(250),
mobile varchar(50),
email varchar(100),
apply_date date,
tranit varchar(150),
status varchar(250),
file_location` VARCHAR(250),
PRIMARY key (id)
)


CREATE TABLE tb_cartificate_main (
id int(11) AUTO_INCREMENT,
reg_number varchar(150) UNIQUE,
name varchar(250),
father_name varchar(250),
address varchar(250),
mobile varchar(50),
email varchar(100),
apply_date date,
tranit varchar(150),
status varchar(250),
session varchar(150),
program varchar(150),
CGPA float,
file_location` VARCHAR(250),
PRIMARY key (id)
)
